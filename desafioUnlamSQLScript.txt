CREATE DATABASE IF NOT EXISTS DESAFIO_UNLAM;

USE DESAFIO_UNLAM;

CREATE TABLE ROL(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(30) NOT NULL
);

CREATE TABLE SEXO(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE PAIS(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(60) NOT NULL
);

CREATE TABLE CATEGORIA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    color VARCHAR(20) NOT NULL
);

CREATE TABLE ESTADO_PREGUNTA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(20) NOT NULL
);

CREATE TABLE RESULTADO(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(15) NOT NULL
);

CREATE TABLE TIPO_PARTIDA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(20) NOT NULL
);

CREATE TABLE CIUDAD(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(60) NOT NULL,
    id_pais INT NOT NULL,
    FOREIGN KEY (id_pais) REFERENCES PAIS (id)
);

CREATE TABLE USUARIO(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre_completo VARCHAR(100) NOT NULL,
    anio_nacimiento INT NOT NULL,
    correo_electronico VARCHAR(100) UNIQUE NOT NULL,
    contrasenia VARCHAR(10) NOT NULL,
    nombre_usuario VARCHAR(10) UNIQUE NOT NULL,
    foto VARCHAR(255) DEFAULT NULL,
    puntos_totales INT NOT NULL DEFAULT 0,
    id_sexo INT NOT NULL,
    id_rol INT NOT NULL,
    id_ciudad INT NOT NULL,
	cuenta_verificada BIT DEFAULT 0,
    codigo_verificacion VARCHAR(255) DEFAULT NULL,
    FOREIGN KEY (id_sexo) REFERENCES SEXO (id),
    FOREIGN KEY (id_rol) REFERENCES ROL (id),
    FOREIGN KEY (id_ciudad) REFERENCES CIUDAD (id)
);

CREATE TABLE PARTIDA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    fecha DATETIME NOT NULL,
    puntaje_jugador1 INT NOT NULL,
    puntaje_jugador2 INT NULL DEFAULT 0,
    id_jugador1 INT NOT NULL,
    id_jugador2 INT NULL,
    id_resultado INT NOT NULL,
    id_tipo_partida INT NOT NULL,
    FOREIGN KEY (id_jugador1) REFERENCES USUARIO (id),
    FOREIGN KEY (id_jugador2) REFERENCES USUARIO (id),
    FOREIGN KEY (id_resultado) REFERENCES RESULTADO (id),
    FOREIGN KEY (id_tipo_partida) REFERENCES TIPO_PARTIDA (id)
);

CREATE TABLE RESPUESTA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    opcion_a VARCHAR(50) NOT NULL,
    opcion_b VARCHAR(50) NOT NULL,
    opcion_c VARCHAR(50) NOT NULL,
    opcion_d VARCHAR(50) NOT NULL,
    respuesta_correcta CHAR(1) NOT NULL
);

CREATE TABLE PREGUNTA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    enunciado VARCHAR(100) NOT NULL,
    id_respuesta INT NOT NULL,
    id_categoria INT NOT NULL,
    id_estado_pregunta INT NOT NULL,
    id_autor INT NOT NULL,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    dificultad_actual ENUM('facil', 'media', 'dificil') NOT NULL DEFAULT 'media',
	respuestas_totales INT NOT NULL DEFAULT 0,
	respuestas_correctas INT NOT NULL DEFAULT 0,
	ratio_aciertos DECIMAL(5,2) NOT NULL DEFAULT 0.00,
    FOREIGN KEY (id_respuesta) REFERENCES RESPUESTA (id),
    FOREIGN KEY (id_categoria) REFERENCES CATEGORIA (id),
    FOREIGN KEY (id_estado_pregunta) REFERENCES ESTADO_PREGUNTA (id),
    FOREIGN KEY (id_autor) REFERENCES USUARIO (id)
);

CREATE TABLE RESPUESTA_USUARIO(
    id INT PRIMARY KEY AUTO_INCREMENT,
    fue_correcta BIT NULL,
    opcion_elegida CHAR(1) NULL,
    id_partida INT NOT NULL,
    id_pregunta INT NOT NULL,
    id_usuario INT NOT NULL,
    FOREIGN KEY (id_partida) REFERENCES PARTIDA (id),
    FOREIGN KEY (id_pregunta) REFERENCES PREGUNTA (id),
    FOREIGN KEY (id_usuario) REFERENCES USUARIO (id)
);

CREATE TABLE AUDITORIA_PREGUNTA(
    id INT PRIMARY KEY AUTO_INCREMENT,
    motivo VARCHAR(100) NOT NULL,
    fecha DATETIME NOT NULL,
    id_solicitante INT NOT NULL,
    id_pregunta INT NOT NULL,
    id_estado_pregunta INT NOT NULL,
    FOREIGN KEY (id_solicitante) REFERENCES USUARIO (id),
    FOREIGN KEY (id_pregunta) REFERENCES PREGUNTA (id),
    FOREIGN KEY (id_estado_pregunta) REFERENCES ESTADO_PREGUNTA (id)
);
